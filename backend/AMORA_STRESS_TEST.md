class FallbackTierSystem:
    """Provides contextual fallbacks based on failure type."""
    
    @staticmethod
    def get_fallback(failure_type, emotional_signals, context):
        if failure_type == "low_similarity":
            # No template matched well
            if emotional_signals.get("confusion", 0) > 0.6:
                return "I want to make sure I understand what you're asking. Can you tell me more about what's going on?"
            elif emotional_signals.get("anxiety", 0) > 0.6:
                return "I hear that something's weighing on you. Help me understand what you're dealing with?"
            else:
                return "I'm listening. What would be most helpful to talk through right now?"
        
        elif failure_type == "unclear_intent":
            # Multiple strong intents, can't disambiguate
            return "I'm hearing a few different things here. What feels most important to focus on right now?"
        
        elif failure_type == "empty_input":
            # User sent very short/empty message
            if context.turns_count > 0:
                return "Take your time. I'm here when you're ready."
            else:
                return "I'm here whenever you're ready to talk. What's on your mind?"
        
        elif failure_type == "emotional_overload":
            # Too many high-intensity emotions
            return "It sounds like you're feeling a lot at once. What's the strongest feeling right now?"
        
        else:
            # System error or unknown
            return "I'm here to help. Can you share a bit more about what you're thinking?"

def adaptive_first_turn(question, emotional_signals):
    """Adapt first-turn response to user's arrival state."""
    
    # Detect arrival state
    urgency = detect_urgency(question)  # "just", "need to", "help"
    formality = detect_formality(question)  # "hi", "hello" vs problem statement
    emotional_intensity = max(emotional_signals.values())
    
    if urgency > 0.7 or emotional_intensity > 0.7:
        # High urgency or emotion: Brief supportive, immediate availability
        responses = [
            "I'm here. What's going on?",
            "I'm listening. Tell me what happened.",
            "You're in the right place. What's going on?"
        ]
        return random.choice(responses)
    
    elif formality < 0.3:
        # Casual greeting ("hi"): Full intro
        return full_welcome_greeting()
    
    elif len(question.split()) > 10:
        # Long first message: Acknowledge + focus
        return f"I hear you. Let's unpack this together. {extract_main_question(question)}"
    
    else:
        # Standard: Warm but brief
        responses = [
            "I'm Amora. I'm here to help you think through this. What's on your mind?",
            "I'm here. What would you like to talk about?",
            "Tell me what's going on."
        ]
        return random.choice(responses)

def check_for_recalibration_need(conversation_state):
    """Determine if mid-session recalibration is needed."""
    
    # After 8+ turns
    if conversation_state.turns_count >= 8:
        # Check for topic drift
        if len(set(conversation_state.recent_themes)) > 3:
            return {
                "type": "refocus",
                "message": "We've covered a lot. What feels most important to focus on?"
            }
        
        # Check for emotional shift
        if conversation_state.emotional_journey:
            start = conversation_state.emotional_journey[0]
            current = conversation_state.emotional_journey[-1]
            if abs(start['intensity'] - current['intensity']) > 0.4:
                return {
                    "type": "progress_check",
                    "message": "How are you feeling about all of this now compared to when we started?"
                }
    
    # After 10+ turns
    if conversation_state.turns_count >= 10:
        # Energy check
        return {
            "type": "energy_check",
            "message": "We've been at this for a bit. Are you getting what you need, or should we shift focus?"
        }
    
    return None

### Refinement 8: Conversation Closure Signals

**Target**: Recognize closure signals and provide wrap-up

**Implementation**:
```python
CLOSURE_SIGNALS = [
    "thank you", "thanks", "this helped", "i feel better",
    "i know what to do", "i think i'm good", "i got it",
    "appreciate it", "that makes sense now"
]

def detect_closure_signal(question):
    return any(signal in question.lower() for signal in CLOSURE_SIGNALS)

def generate_closure_response(conversation_state, emotional_journey):
    """Provide wrap-up when user signals completion."""
    
    # Reflect on progress
    if len(emotional_journey) > 5:
        start_emotion = emotional_journey[0]['emotion']
        end_emotion = emotional_journey[-1]['emotion']
        
        if start_emotion != end_emotion:
            progress_note = f"You came in feeling {start_emotion}, and you sound {end_emotion} now. That's real movement."
        else:
            progress_note = "You've done some important thinking here."
    else:
        progress_note = "I'm glad this helped."
    
    # Empowerment
    empowerment = random.choice([
        "You figured this out yourself.",
        "You already had the answers.",
        "Trust yourself to handle this.",
        "You know what you need to do."
    ])
    
    # Optional return invitation
    return_invite = "I'm here if you need to talk through anything else."
    
    return f"{progress_note} {empowerment} {return_invite}"
```
